This is Utiaji which inherits from Utiaji::App.
To run another app, change the portion in bold in bin/utiaji : 

#!/usr/bin/env perl6

use lib 'lib';

use Utiaji::Server;
use Utiaji::Log;
use Utiaji;

my $server = Utiaji::Server.new(host => 'localhost', port => 3333);
$server.app = Utiaji.new;
$server.start;
$server.await;
e.g. 
#!/usr/bin/env perl6

use lib 'lib';

use Utiaji::Server;
use Utiaji::Log;
use Utiaji::App::Pim;

my $server = Utiaji::Server.new(host => 'localhost', port => 3333);
$server.app = Utiaji::App::Pim.new;
$server.start;
$server.await;
